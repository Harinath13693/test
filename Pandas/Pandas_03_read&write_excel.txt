import pandas as pd 
Reading excel file to dataframe
df=pd.read_excel('path_of_file','sheet_name',converters={columnname: function_name})
--converters
  helps to call function for each value in the mentioned column
  converters={columnname1: function_name1,
              columnname2: function_name2
              ............
              columnnameN: function_namen}
